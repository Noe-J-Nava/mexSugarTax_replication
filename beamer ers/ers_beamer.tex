\subsection[Lit. Review]{Literature Review}
\frame{% Literature review on 
\frametitle{Non-demand estimation approaches to study the impact of the Sugary Soft Beverages}
	
	\begin{itemize}
		\item Passed-through effect:
			\begin{itemize}
				\item Colchero \textit{et al.,} (2015b): \textbf{Tax was almost completely absorbed by consumers}.
				\item Campos-Vázquez and Medina-Cortina (2019): After accounting for competition at the supermarket level, the consumer experienced price changes \textbf{between 24\% and 152\% pesos} of the tax value.
			\end{itemize}
		\item SSB sales effect:
			\begin{itemize}
				\item Colchero \textit{et al.,} (2016): \textbf{7.3\% decrease} in per capital sales of SSB.
			\end{itemize}
		\item Caloric intake decrease:
			\begin{itemize}
				\item Aguilar, Gutierrez, and Seira (2021): \textbf{2.7\% decrease} in caloric intake.
			\end{itemize}
	\end{itemize}

}

\frame{% Caveats
\frametitle{Caveats:}
	
	\begin{itemize}
		\item Passed-through and sales studies suggest a clear decline in the consumption of SSB;
			\begin{itemize}
				\item However, \textbf{substitution could take place} for caloric-intensive goods.
			\end{itemize}
		\item Causal impact of the tax on caloric intake suggests a decline on obesity;
			\begin{itemize}
				\item However, we do not understand the \textbf{mechanisms} (unintended consequences):
				\begin{itemize}
					\item Budget effect.
					\item Substitution and complementarity effects.
				\end{itemize}
			\end{itemize}
	\end{itemize}

}

\frame{% Literature review on SSB demand estimations in Latin-america
	
	\frametitle{Demand estimation approaches in Latin America}

	\begin{itemize}
		\item In México, Colchero, Salgado, and Unar-Munguía (2015a): Own-price elasticity at \textbf{-1.11}.
		\item In Colombia, Caro \textit{et al.,} (2017): Own-price elasticity at \textbf{-1.61}.
		\item In Chile, Guerrero-López, Unar-Munguía, and Colchero (2017): Own-price elasticity at \textbf{-1.37}.
		\item All SSB demand estimation studies in Latin-america have at least two of the following empirical shortcomings:
				\begin{itemize}
					\item[1] \textbf{Unit-value endogeneity} (Deaton, 1998; Cox and Wohlgenant, 1986).
					\item[2] \textbf{Expediture endogeneity} (LeFrance, 1993).
					\item[3] \textbf{Corner solutions treatment} (Dong, Gould, and Kaiser, 2004; Enríquez and Echevarría, 2016).
				\end{itemize}
	\end{itemize}

}

\section{Empirical strategy}
\subsection[QUAI Demand System]{QUAI Demand System}

\frame{% Banks, Blundell and Lewbel (1997)'s QUAI Demand System
	
	\label{QUAIDS}

	\frametitle{Banks, Blundell and Lewbel (1997)'s QUAI Demand System}
	%\footnotesize
	
\begin{equation}
	U_1 = \alpha_1 + \mathbf{\theta'_{1}z}  + \sum^{M}_{j = 1} \gamma_{1j} \ln p_j + \beta_1\ln \Big\{ \frac{w}{a(\mathbf{p})} \Big\} + \frac{\lambda_1}{b(\mathbf{p})} \left[ \ln \Big\{ \frac{w}{a(\mathbf{p})} \Big\}\right]^2_{,}
	\nonumber
\end{equation}

\begin{equation}
	\vdots
	\nonumber
\end{equation}
	
\begin{equation}
	U_M = \underbrace{\alpha_M + \mathbf{\theta'_{M}z}  + \sum^{M}_{j = 1} \gamma_{Mj} \ln p_j + \beta_M\ln \Big\{ \frac{w}{a(\mathbf{p})} \Big\}}_\text{AID system} + \underbrace{\frac{\lambda_M}{b(\mathbf{p})} \left[ \ln \Big\{ \frac{w}{a(\mathbf{p})} \Big\}\right]^2_{.}}_\text{QUAID system}
	\nonumber
\end{equation}

\begin{beamercolorbox}{coloredboxstuff}

where:
\begin{itemize}
	\fontsize{6pt}{7.2}\selectfont
	\item $\mathbf{z}$ and $w$ are a vector of demographic characteristics and a total budget, respectively.
	\item $a(\mathbf{p})$ and $b(\mathbf{p})$ are priced aggregators.
	\item $p_i$ is a vector of prices $\forall i$.
	\item The dependent variable is defined as $U_i = \frac{p_i q_i}{w}$ $\forall i$ where $q_i$ is quantity . 
	
	\item The coefficients are restricted such that: \\
		$\sum^{M}_{i = 1} \alpha_i = 1$, $\sum^{M}_{i = 1} \beta_i = 0$, $\sum^{M}_{j = 1} \gamma_{ij} = 0$, $\sum^{M}_{i = 1} \lambda_i = 0$, $\sum^{M}_{i = 1} \theta_i = 0$, and $\gamma_{ij} = \gamma_{ji}$  $\forall i,j$.
\end{itemize}

\end{beamercolorbox}

}

\frame{% Engel and Cournot aggregations

	%\footnotesize
	\frametitle{Cournot and Engel aggregations}

The budget restriction is given by:
	\begin{equation}
		p_1 q_1 + \hdots + p_M q_M = w
		\nonumber 
	\end{equation}

The Cournot aggregation is given by:
	\begin{equation}
		U_1 \eta_{1j} + \hdots + U_M \eta_{Mj} = - U_j
		\nonumber
	\end{equation}

The Engel aggregation is given by:
	\begin{equation}
		U_1 \eta_1 + \hdots + U_M \eta_M = 1
		\nonumber
	\end{equation}

\begin{beamercolorbox}{coloredboxstuff}
where:
\begin{itemize}
	\item $\eta_i$ is a budget elasticity.
	\item $\eta_{mj}$ is a price elasticity.
\end{itemize}
\end{beamercolorbox}

}

\frame{% Engel and Cournot aggregations

	%\footnotesize
	\frametitle{The Slutsky Equation}

Slutsky equation:
	\begin{equation}
		\frac{\partial h_i}{\partial p_j} = \frac{\partial x_i}{\partial p_j} + \frac{\partial x_i}{\partial w} x_j
		\nonumber 
	\end{equation}

Slutsky equation in elasticities:
\begin{equation}
	\eta^c_{ij} = \eta_{ij} + \eta_i U_j
	\nonumber 
\end{equation}

\begin{beamercolorbox}{coloredboxstuff}
	where:
	\begin{itemize}
		\item $h_i$ is the Hicksian demand.
		\item $\eta^c_{ij}$ is the Hicksian elasticity.
	\end{itemize}
\end{beamercolorbox}
}
